#Task 2, 3, 4, 5
def display_blanks():
  """
  This function will be called over and over again to display the correctly-guessed letters. Letters not guessed correctly will be displayed with an underscore symbol.
  """
  #Task 3 - define empty string for display_txt
  display_txt = ""
  
  #Task 4 - loop through each letter of the answer and add the correct letters or underscore
  for letter in answer:
    if letter in chosen_letters:
      display_txt += letter + " "
    else:
      display_txt += "_ "
  
  #Task 5 - return the complete display_txt
  return display_txt + "\n"
  
#print the blank spaces based on the length of
#the answer when the game starts  
print("_ " * len(answer))
